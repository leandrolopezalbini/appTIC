function validarPayload(data, camposRequeridos) {
  if (!data || typeof data !== 'object') {
    return { valid: false, error: 'Payload debe ser un objeto válido' };
  }
  for (let campo of camposRequeridos) {
    if (!data.hasOwnProperty(campo) || data[campo] === null || data[campo] === '') {
      return { valid: false, error: `Campo requerido faltante: ${campo}` };
    }
  }
  return { valid: true, error: null };
}

let CACHE_SHEETS = {};
let CACHE_TIMESTAMPS = {};
const CACHE_TTL = 5 * 60 * 1000;

function precargarCacheHojas() {
  const ahora = Date.now();
  const cacheExpired = (key) => !CACHE_TIMESTAMPS[key] || (ahora - CACHE_TIMESTAMPS[key]) > CACHE_TTL;
  try {
    if (cacheExpired('TABLEROS')) {
      CACHE_SHEETS['TABLEROS'] = OBTENER_HOJA(NOMBRE_HOJA.TABLEROS).getDataRange().getValues();
      CACHE_TIMESTAMPS['TABLEROS'] = ahora;
      Logger.log("✅ Caché TABLEROS actualizado");
    }
    if (cacheExpired('CAMARAS')) {
      CACHE_SHEETS['CAMARAS'] = OBTENER_HOJA(NOMBRE_HOJA.CAMARAS).getDataRange().getValues();
      CACHE_TIMESTAMPS['CAMARAS'] = ahora;
      Logger.log("✅ Caché CAMARAS actualizado");
    }
    if (cacheExpired('INSUMOS')) {
      CACHE_SHEETS['INSUMOS'] = OBTENER_HOJA(NOMBRE_HOJA.INSUMOS).getDataRange().getValues();
      CACHE_TIMESTAMPS['INSUMOS'] = ahora;
    }
    if (cacheExpired('INSTALACIONES')) {
      CACHE_SHEETS['INSTALACIONES'] = OBTENER_HOJA(NOMBRE_HOJA.INSTALACIONES).getDataRange().getValues();
      CACHE_TIMESTAMPS['INSTALACIONES'] = ahora;
    }
    if (cacheExpired('ACTIVOS')) {
      CACHE_SHEETS['ACTIVOS'] = OBTENER_HOJA(NOMBRE_HOJA.ACTIVOS).getDataRange().getValues();
      CACHE_TIMESTAMPS['ACTIVOS'] = ahora;
    }
    if (cacheExpired('SOLICITUDES')) {
      CACHE_SHEETS['SOLICITUDES'] = OBTENER_HOJA(NOMBRE_HOJA.SOLICITUDES).getDataRange().getValues();
      CACHE_TIMESTAMPS['SOLICITUDES'] = ahora;
    }
    Logger.log("✅ Caché de hojas precargado completamente");
    return { success: true, cached: Object.keys(CACHE_SHEETS).length };
  } catch (e) {
    Logger.log("❌ Error precargando caché: " + e.message);
    CACHE_SHEETS = {};
    return { success: false, error: e.message };
  }
}

function limpiarCache() {
  CACHE_SHEETS = {};
  CACHE_TIMESTAMPS = {};
  Logger.log("✅ Caché limpiado");
}

function testTrazabilidad() {
  const resultado = [];
  const hoja = OBTENER_HOJA(NOMBRE_HOJA.HISTORIAL);
  if (!hoja) {
    resultado.push("❌ FALLO: No existe hoja Historial");
    return resultado;
  }
  const data = hoja.getDataRange().getValues();
  if (data.length <= 1) {
    resultado.push("❌ FALLO: Historial está vacío");
    return resultado;
  }
  const ultimaFila = data[data.length - 1];
  const headers = data[0];
  const camposRequeridos = ['Fecha', 'Tipo_Evento', 'ID_Activo', 'Usuario', 'Detalles'];
  let todosPresentes = true;
  for (let campo of camposRequeridos) {
    const idx = getHeaderIndex(headers, campo);
    if (idx < 0 || !ultimaFila[idx]) {
      resultado.push("⚠️ ADVERTENCIA: Campo " + campo + " está vacío");
      todosPresentes = false;
    }
  }
  if (todosPresentes) {
    resultado.push("✅ PASÓ: Historial tiene registros completos");
    resultado.push("📊 Total registros: " + (data.length - 1));
  }
  return resultado;
}

function notificarEvento(tipoEvento, detalles) {
  const resultado = {
    success: true,
    enviados: [],
    fallos: [],
    total: 0
  };

  try {
    const emails = obtenerDestinatariosNotificacion(tipoEvento);
    if (!emails || emails.length === 0) {
      resultado.success = false;
      resultado.error = "No hay destinatarios configurados para: " + tipoEvento;
      Logger.log("⚠️ " + resultado.error);
      return resultado;
    }

    const asunto = generarAsuntoNotificacion(tipoEvento);
    const cuerpo = generarCuerpoNotificacion(tipoEvento, detalles);

    resultado.total = emails.length;

    for (let email of emails) {
      try {
        MailApp.sendEmail(email, asunto, cuerpo);
        resultado.enviados.push(email);
        Logger.log("✅ Email enviado a: " + email);
      } catch (emailError) {
        resultado.fallos.push({
          email: email,
          error: emailError.message
        });
        Logger.log("❌ Fallo al enviar a " + email + ": " + emailError.message);
      }
    }

    if (resultado.fallos.length === 0) {
      resultado.success = true;
      Logger.log("✅ Notificación " + tipoEvento + " enviada completamente");
    } else if (resultado.enviados.length > 0) {
      resultado.success = true;
      Logger.log("⚠️ Notificación enviada parcialmente (" + resultado.enviados.length + "/" + resultado.total + ")");
    } else {
      resultado.success = false;
      Logger.log("❌ Notificación " + tipoEvento + " falló completamente");
    }

    return resultado;

  } catch (e) {
    Logger.log("❌ Error crítico en notificarEvento: " + e.message);
    return {
      success: false,
      error: e.message,
      enviados: [],
      fallos: emails || []
    };
  }
}
